    <script>
        // Import the PHP data directly into a JavaScript object
        const appData = <?php echo $jsonData; ?>;
        
        // Track current game for the generator
        let currentGame = 'powerball';

        // Function to render mini balls in the sidebar
        function renderMiniBalls(numbers, type) {
            return numbers.map(num => `<li class="mini-ball ${type}">${num}</li>`).join('');
        }

        // Function to render pairs in the sidebar
        function renderPairs(pairs) {
            return pairs.map(pair => `<span class="pair-badge">${pair}</span>`).join('');
        }

        // Main Tab Switching & SPA Logic
        function switchTab(gameId) {
            currentGame = gameId;
            const data = appData[gameId];

            // 1. Update Tabs UI
            document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));
            document.querySelectorAll('.tab-content').forEach(content => {
                content.classList.remove('active');
                content.style.opacity = '0'; // For smooth fade
            });
            
            // Find the clicked button and set it active
            const activeBtn = Array.from(document.querySelectorAll('.tab-btn')).find(
                btn => btn.textContent.trim().toLowerCase().includes(gameId.replace('plus', ' Plus'))
            );
            if (activeBtn) activeBtn.classList.add('active');
            
            const targetContent = document.getElementById(gameId + '-content');
            targetContent.classList.add('active');
            setTimeout(() => targetContent.style.opacity = '1', 50);

            // 2. Update Dynamic History Link at the bottom of the left column
            const historyLink = document.getElementById('dynamic-history-link');
            historyLink.href = `/history/${gameId}`;
            historyLink.innerText = `View Full ${data.name} History`;

            // 3. Update the Right Sidebar Headers
            document.getElementById('dynamic-sidebar-label').innerText = data.name;
            document.getElementById('dyn-gen-label').innerText = data.name;

            // 4. Update the Analytics Data (Hot/Cold/Pairs)
            document.getElementById('dyn-hot-numbers').innerHTML = renderMiniBalls(data.analytics.hot, 'hot');
            document.getElementById('dyn-cold-numbers').innerHTML = renderMiniBalls(data.analytics.cold, 'cold');
            document.getElementById('dyn-frequent-pairs').innerHTML = renderPairs(data.analytics.pairs);
            
            // 5. Reset Generator Display
            document.getElementById('generated-numbers').innerHTML = '-- -- -- -- --';
        }

        // Context-Aware Smart Generator Logic
        function generateTicket() {
            const rules = appData[currentGame].rules;
            const resultDiv = document.getElementById('generated-numbers');
            
            // Loading state
            resultDiv.innerHTML = '<span style="font-size: 1rem; color: #94a3b8; letter-spacing: 0;">Calculating parameters...</span>';
            resultDiv.style.opacity = '0.5';

            setTimeout(() => {
                let numbers = [];
                // Generate main balls based on game specific rules
                while(numbers.length < rules.draw_size){
                    let r = Math.floor(Math.random() * rules.max_num) + 1;
                    if(numbers.indexOf(r) === -1) numbers.push(r);
                }
                numbers.sort((a,b) => a - b);
                
                // Format main numbers
                let formatted = numbers.map(n => n.toString().padStart(2, '0')).join(' &nbsp; ');
                
                // Append Bonus Ball if the game requires it
                if(rules.has_bonus) {
                    let bonusNum = Math.floor(Math.random() * rules.bonus_max) + 1;
                    formatted += ` <span style="color: white; margin: 0 10px;">|</span> <span style="color: #fbbf24;">${bonusNum.toString().padStart(2, '0')}</span>`;
                }
                
                resultDiv.innerHTML = formatted;
                resultDiv.style.opacity = '1';
            }, 500); 
        }

        // Initialize the sidebar on first load
        window.onload = () => {
            switchTab('powerball'); 
        };
    </script>
